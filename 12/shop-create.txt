## 1.프로젝트 생성
nest new shop --skip-git

cd shop 

## 2.모듈 생성 --REST API > Yes 
nest g resource categories --no-spec
nest g resource products --no-spec
nest g resource users --no-spec

## 3.패키지 설치
npm i class-validator class-transformer dotenv
npm i @prisma/client@6
npm i -D prisma@6
npm i @nestjs/swagger 

## prisma generate
npx prisma init

## schema.prisma 수정 
  1)generator client {
    provider = "prisma-client-js"
  }
  2)model 생성 

## schema.prisma 파일을 정렬 깨끗하게.
npx prisma format 

    // DB  cli로 진행할 경우 명령어 start  (psql install 필요)
    psql -U postgres postgres;
    create database shop;
    grant all PRIVILEGES on database shop to postgres;
    \c shop
    \l  #데이타베이스조회
    \d  #테이블보기 
    \d 테이블명 #테이블 상세보기 
    grant all on schema public to postgres;
    // cli로 진행할 경우 명령어 end
